def ChatGPT(self):
        os.system("cls")

        # Coleta chave e define
        Api_Key = arq.ValoresDeChaves()
        openai.api_key = Api_Key

        # Input do usuÃ¡rio
        prompt_usuario = input("Descreva o estilo desejado (ex: estilo anime): ")
        caminho_imagem = input("Caminho da imagem (ex: ./foto.png): ")

        # Cria assistente
        assistant = openai.beta.assistants.create(
            name="Assistente com visÃ£o",
            instructions="VocÃª Ã© um assistente que descreve imagens com base no estilo desejado pelo usuÃ¡rio.",
            tools=[],
            model="gpt-4-vision-preview"
        )

        # Cria thread
        thread = openai.beta.threads.create()

        # Upload da imagem
        image_file = openai.files.create(
            file=open(caminho_imagem, "rb"),
            purpose="assistants"
        )

        # Envia imagem + estilo desejado
        openai.beta.threads.messages.create(
            thread_id=thread.id,
            role="user",
            content=[
                {"type": "text", "text": f"Descreva essa imagem como se fosse no estilo: {prompt_usuario}"},
                {"type": "image_file", "image_file": {"file_id": image_file.id}}
            ]
        )

        # Roda o assistente
        run = openai.beta.threads.runs.create(
            thread_id=thread.id,
            assistant_id=assistant.id
        )

        print("Aguardando anÃ¡lise da imagem...")
        while True:
            run_status = openai.beta.threads.runs.retrieve(
                thread_id=thread.id,
                run_id=run.id
            )
            if run_status.status == "completed":
                break
            time.sleep(1)

        # Captura resposta
        messages = openai.beta.threads.messages.list(thread_id=thread.id)
        descricao = ""
        for msg in messages.data:
            if msg.role == "assistant":
                descricao = msg.content[0].text.value
                break

        print("\nDescriÃ§Ã£o gerada para o DALLÂ·E:\n", descricao)

        # GeraÃ§Ã£o da imagem com DALLÂ·E
        response = openai.Image.create(
            prompt=descricao,
            n=1,
            size="1024x1024"
        )

        url_imagem = response['data'][0]['url']
        print("\nâœ… Imagem gerada com sucesso!")
        print("ğŸ”— Link da imagem:", url_imagem)
